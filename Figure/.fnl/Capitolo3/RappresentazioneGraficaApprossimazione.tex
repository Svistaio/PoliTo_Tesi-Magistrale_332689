        \newcommand\drawCompleteEdges[1]{
            \begin{scope}[on background layer]
                \foreach \i [evaluate=\j as \s using int(\i+1)] in {1,...,6}
                    \foreach \j in {\s,...,7}{
                        \ifcase\i % Caso 0

                            \fi\fi\fi\fi\fi
                        \or % Caso 3
                            \ifnum\j=5\def\b{-20}
                            \else\ifnum\j=7\def\b{-10}
                            \else\def\b{20}
                            \fi\fi
                        \or % Caso 4
                            \def\b{20}
                        \or % Caso 5
                            \def\b{-20}
                        \or % Caso 6
                            \def\b{20}

                            \def\b{20}

                        \ifnum\i=1

                        \else
                            \draw[nodeArrowApprox,transparentNodeArrow] (#1\i) to[bend left=\b] (#1\j);
                        \fi
                    }
            \end{scope}
