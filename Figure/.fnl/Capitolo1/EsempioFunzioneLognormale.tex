\begin{tikzpicture}
    \infopageSharedKeys

    %\begingroup Parametri
        \renewcommand\labelSize{10}
        \renewcommand\tickSize9
        \renewcommand\legendSize9
        \renewcommand\subCaptionSize9
    
        \newcommand\width{10cm}
        \pgfmathsetlengthmacro\height{0.618*\width}
    %\endgroup

    %\begingroup Funzioni
        \pgfmathsetmacro\mean{10}
        \pgfmathsetmacro\lasts{0.04}
        \newcommand\lognormalX{(1/(x*log(10)*s*sqrt(2*pi)))*exp(-(log10(x)-m)**2/(2*s**2))}
        \newcommand\normal{(1/(s*sqrt(2*pi)))*exp(-(x-m)**2/(2*s**2))}

        \newcommand\plotFunction[2]{
            \foreach \s [evaluate=\s as \m using log10(\mean)-\s^2*ln(10)/2] in {%
                0.74,0.63,0.45,0.25,0.15,0.08,0.05,\lasts
            }{%
                % \pgfmathsetmacro\m{1-\s^2*ln(10)/2}
                \addplot+ [thick,no marks] gnuplot [raw gnuplot] {%
                    set samples 5000; s=\s; m=\m; #1
                };#2
                \addlegendentryexpanded{
                    \makebox[5.5em][l]{
                        \(\sigma=\pgfmathprintnumber[
                            fixed,
                            precision=3,
                            fixed zerofill
                        ]{\s}\)
                    }\(\mu=\pgfmathprintnumber[
                        fixed,
                        precision=3,
                        fixed zerofill
                    ]{\m}\)
                }
            }
            % Parametri validi per la lognormale naturale con «ln(x)»
            % \foreach \s in {2.5,2,1.5,1,0.5,0.3,0.2,0.13,0.1}{%
            %     \addplot+ [thick,no marks] gnuplot [raw gnuplot]{%
            %         set samples 5000; m=\m; s=\s*log10(exp(1)); #1
            %     };
            %     \pgfmathparse{round(\s*log10(exp(1))*1000)/1000}
            %     \addlegendentryexpanded{\(\sigma=\pgfmathresult\)}
            % }
        }
        \newcommand\plotLognormalAverage{
            \addplot [draw=none,forget plot,dashed] gnuplot [raw gnuplot] {%
                f(x,m,s)=\lognormalX; m=\m; s=\s; plot[x=\mean:\mean] (f(x,m,s));
            } coordinate (media); 
            \pgfplotsset{cycle list shift=-1}
            \addplot+ [draw=.,dashed,thick,forget plot] (\mean,0) -- (media); 

            % \addplot [draw=.,forget plot,dashed] gnuplot [raw gnuplot] {%
            %     f(x,m,s)=\lognormalX; m=\m; s=\s;
            %     plot[x=10**\m:10**\m] (f(x,m,s));
            % } -- (axis cs:{10^\m},0); 
            % \draw [line width=.5pt,dashed,draw=.] (axis cs:{10^\m},0) -- (media);
        }
        \newcommand\plotNormalAverage{
            \addplot [draw=none,forget plot,dashed] gnuplot [raw gnuplot] {%
                f(x,m,s)=\normal; m=\m; s=\s; plot[x=\m:\m] (f(x,m,s));
            } coordinate (media); 
            \pgfplotsset{cycle list shift=-1}
            \addplot+ [draw=.,dashed,thick,forget plot] (\m,0) -- (media); 

            % \addplot [draw=none,forget plot] gnuplot [raw gnuplot] {%
            %     f(x,m,s)=\normal; m=\m; s=\lasts; plot[x=\m:\m] (f(x,m,s));
            % } coordinate (media); 
            % \draw [line width=.5pt,dashed,black] (axis cs:\m,0) -- (media);
        }
    %\endgroup

    %\begingroup Impostazioni
        \pgfplotsset{
            %\begingroup Gruppi
            /pgfplots/group/goupStyle/.style={
                group name=Lognormale,
                vertical sep=2.5em,
                % x descriptions at=edge bottom,
                y descriptions at=edge left,
                group size=1 by 3,
                % group size=1 by 1,
            },
            %\endgroup
            %
            %\begingroup Grafici
            plotStyle/.style={
                width=\width,
                height=\height,
                % x post scale=0.7,
                % y post scale=0.8,
                scale only axis,
                axis lines=left,
                xlabel style={font=\dimTesto{\labelSize}\color{white!15!black}},
                ylabel style={font=\dimTesto{\labelSize}\color{white!15!black}},
                xticklabel style={font=\dimTesto{\tickSize}},
                yticklabel style={font=\dimTesto{\tickSize}},
                grid=major,
                grid style={
                    dashed,
                    gray!20
                },
                colormap/viridis,
                cycle list={[samples of colormap={8} of viridis]},
            },
            label0/.style={ylabel={\(f_X(x)\)}},
            label1/.style={xlabel={\(x\)},label0},
            label2/.style={xlabel={\(y\)},ylabel={\(f_{Y}(y)\)}},
            yboundLognormal/.style={ymin=0,ymax=0.45},
            yboundNormal/.style={ymin=0,ymax=10}
            %\endgroup
        }
    %\endgroup

    \begin{groupplot}[
        group style={goupStyle},
        plotStyle,
        enlarge y limits={upper,value=.01},
    ]
        %\begingroup
            \gdef\xmin{1e-10}\gdef\xmax{15}
            \nextgroupplot[%
                legend style={
                    \placeLegend{0}{1}{north west}{3em}{0em},
                    legend cell align=left
                },
                label0,
                xlabel={},
                xmode=linear,
                yboundLognormal,
                xmin=\xmin,xmax=\xmax,
            ]\plotFunction{
                f(x,m,s)=\lognormalX; plot [x=\xmin:\xmax] f(x,m,s);
            }\plotLognormalAverage
            %\legend{}
        %\endgroup

        %\begingroup
            \gdef\xmin{5e-3}\gdef\xmax{50}
            \nextgroupplot[%
                legend style={
                    \placeLegend{0}{0}{south west}{0em}{0em},
                    legend cell align=left
                },
                label1,
                xmode=log,
                yboundLognormal,
                xmin=\xmin,xmax=\xmax,
            ]\plotFunction{%
                f(x,m,s)=\lognormalX; set parametric;
                plot [x=-6:6] 10**(m+s*x), f(10**(m+s*x),m,s);
            }\plotLognormalAverage
        %\endgroup

        %\begingroup
            \gdef\xmin{0}\gdef\xmax{1.5}
            \nextgroupplot[%
                legend style={
                    \placeLegend{0}{1}{north west}{0em}{0em},
                    legend cell align=left
                },
                label2,
                yshift=-1em,
                yboundNormal,
                % ymode=log,
                xmin=\xmin,xmax=\xmax,
            ]\plotFunction{%
                f(x,m,s)=\normal;
                set parametric; plot [t=-5:5] m+s*t, f(m+s*t,m,s);
            }\plotNormalAverage
        %\endgroup
    \end{groupplot}

    %\begingroup Didascalie
        \foreach \r in {1,...,3}{
            \writeSubCaption{Lognormale c1r\r}{north east}{north east}{0em}{0em}

            % \else\writeSubCaption{Lognormale c1r\r}{south west}{south west}{0em}{0em}\fi % Caso 3
        } % https://tex.stackexchange.com/questions/632045/get-subcaption-for-every-plot-using-groupplot-pgf-tikz-after-2020-update
    %\endgroup

    \redefineTikZbounds{Lognormale c1r1}{Lognormale c1r3}
\end{tikzpicture}
